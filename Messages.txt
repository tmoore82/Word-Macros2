Public Sub LossAlertNT()
'Version 1
'Beta Release v1.0
'tmoore82
'10/02/12 (undated until now)

'reminds operator that certain things will not be copied in a cleanup

Dim strLost As String

If ActiveDocument.Tables.Count > 0 Then strTables = _
    "There are tables in this document." & vbCr & _
    "They will be converted to unformatted text." & vbCr & vbCr
If ActiveDocument.Footnotes.Count > 0 Then strLost = "     - Footnotes" & vbCr
If ActiveDocument.Endnotes.Count > 0 Then strLost = strLost & "     - Endnotes" & vbCr
If ActiveDocument.Shapes.Count > 0 Or ActiveDocument.InlineShapes.Count > 0 Then _
    strLost = strLost & "     - Graphics and/or Text Boxes"
If Len(strLost) > 0 Then
    strLost = strTables & "The following were found in this document:" & vbCr & _
        strLost & vbCr & _
        "They will NOT be copied to the new document."
Else
    strLost = strTables
End If

If Len(strLost) > 0 Then
    Dim Response
    Response = MsgBox(strLost, 49, "IMPORTANT WARNING") '49=1(vbOKCancel)+48(vbExclamation)
    If Response = 2 Then End '2=Cancel button
End If

End Sub
Public Sub CompletedNT()
'Version 2
'Beta Release v1.0
'tmoore82
'10/02/12 (undated until now)

Dim strChk As String
strChk = "Task completed. Save the document and check for errors."
FillqArray
MsgBox (strChk & vbNewLine & vbNewLine & strQ)
End Sub
